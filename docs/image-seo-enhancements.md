## 2. Structured Data Implementation

### 2.1 ImageObject Schema for Add-on Pages

Implement ImageObject schema for each add-on image to enable rich results in Google Search.

**File Location:** `/storefront/lib/utils/addon-image-schema.ts`

```typescript
/**
 * Generate ImageObject schema for add-on images
 * Based on schema.org/ImageObject
 */

interface AddonImageSchema {
  '@context': string;
  '@type': string;
  contentUrl: string;
  url: string;
  width: number;
  height: number;
  caption: string;
  description: string;
  contentLocation: {
    '@type': string;
    name: string;
    address: {
      '@type': string;
      addressRegion: string;
      addressCountry: string;
    };
    geo?: {
      '@type': string;
      latitude: number;
      longitude: number;
    };
  };
  creditText: string;
  creator: {
    '@type': string;
    name: string;
    url: string;
  };
  license: string;
  acquireLicensePage: string;
  thumbnailUrl?: string;
  encodingFormat: string;
}

/**
 * Generate ImageObject schema for an add-on image
 */
export function generateAddonImageSchema(
  addonHandle: string,
  addonTitle: string,
  imagePath: string,
  altText: string,
  photographer: string,
  photographerUrl: string,
  location: 'Fraser Island' | 'Rainbow Beach' | 'Sunshine Coast'
): AddonImageSchema {
  const baseUrl = 'https://sunshinecoast4wdtours.com.au';
  const imageUrl = `${baseUrl}${imagePath}`;

  // Location-specific coordinates
  const locationData = {
    'Fraser Island': {
      name: 'Fraser Island',
      region: 'Queensland',
      geo: { latitude: -25.2744, longitude: 153.1069 }
    },
    'Rainbow Beach': {
      name: 'Rainbow Beach',
      region: 'Queensland',
      geo: { latitude: -25.9069, longitude: 153.0883 }
    },
    'Sunshine Coast': {
      name: 'Sunshine Coast',
      region: 'Queensland',
      geo: { latitude: -26.6500, longitude: 153.0667 }
    }
  };

  const loc = locationData[location];

  return {
    '@context': 'https://schema.org',
    '@type': 'ImageObject',
    contentUrl: imageUrl,
    url: imageUrl,
    width: 1200,
    height: 800,
    caption: altText,
    description: `${addonTitle} for Fraser Island and Rainbow Beach 4WD tours on Queensland's Sunshine Coast`,
    contentLocation: {
      '@type': 'Place',
      name: loc.name,
      address: {
        '@type': 'PostalAddress',
        addressRegion: loc.region,
        addressCountry: 'AU'
      },
      geo: {
        '@type': 'GeoCoordinates',
        latitude: loc.geo.latitude,
        longitude: loc.geo.longitude
      }
    },
    creditText: `Photo by ${photographer}`,
    creator: {
      '@type': 'Person',
      name: photographer,
      url: photographerUrl
    },
    license: 'https://www.pexels.com/license/',
    acquireLicensePage: 'https://www.pexels.com/license/',
    encodingFormat: 'image/jpeg'
  };
}

/**
 * Get all add-on image schemas
 */
export function getAllAddonImageSchemas(): AddonImageSchema[] {
  const manifest = require('../../../docs/addon-images-manifest.json');

  return manifest.images.map((img: any) => {
    // Determine location from alt text or addon handle
    let location: 'Fraser Island' | 'Rainbow Beach' | 'Sunshine Coast' = 'Fraser Island';
    if (img.alt_text.includes('Rainbow Beach')) {
      location = 'Rainbow Beach';
    } else if (img.alt_text.includes('Sunshine Coast') && !img.alt_text.includes('Fraser Island')) {
      location = 'Sunshine Coast';
    }

    return generateAddonImageSchema(
      img.addon_handle,
      img.addon_title,
      img.image_path,
      img.alt_text,
      img.photographer,
      img.photographer_url,
      location
    );
  });
}
```

## 4. Image Sitemap

### 4.1 Create Image Sitemap

Add image entries to the existing sitemap to help Google discover and index all add-on images.

**File Location:** `/storefront/app/sitemap.ts` (extend existing sitemap)

```typescript
import { MetadataRoute } from 'next';
import addonManifest from '../../docs/addon-images-manifest.json';

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const baseUrl = 'https://sunshinecoast4wdtours.com.au';

  // Existing sitemap entries
  const staticPages = [
    // ... existing pages ...
  ];

  // ADD: Add-ons page with images
  const addonsPage = {
    url: `${baseUrl}/addons`,
    lastModified: new Date(),
    changeFrequency: 'weekly' as const,
    priority: 0.8,
    // Add image information
    images: addonManifest.images.map(img => ({
      url: `${baseUrl}${img.image_path}`,
      title: img.addon_title,
      caption: img.alt_text,
      geoLocation: img.alt_text.includes('Fraser Island')
        ? 'Fraser Island, Queensland, Australia'
        : img.alt_text.includes('Rainbow Beach')
        ? 'Rainbow Beach, Queensland, Australia'
        : 'Sunshine Coast, Queensland, Australia',
      license: img.license_url
    }))
  };

  return [...staticPages, addonsPage];
}
```
